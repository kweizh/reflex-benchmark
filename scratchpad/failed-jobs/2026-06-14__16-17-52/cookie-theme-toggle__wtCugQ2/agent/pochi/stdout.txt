```python
class State(rx.State):
    theme: str = rx.Cookie("light", name="app_theme")

    def toggle_theme(self):
        if self.theme == "light":
            self.theme = "dark"
        else:
            self.theme = "light"

def index() -> rx.Component:
    return rx.container(
        rx.vstack(
            rx.heading(rx.text.span("Current: "), State.theme),
            rx.button("Toggle Theme", on_click=State.toggle_theme),
            ...
        ),
    )
```
